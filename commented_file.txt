name: Go
on:
  pull_request:
    types: [opened, synchronize]

jobs:
  backend:
    runs-on: ubuntu-latest

    steps:
      - uses: actions/checkout@v3
      - name: Setup Go
        uses: actions/setup-go@v4
        with:
          go-version: "1.19"
      - name: Install dependencies
        run: |
          cd backend
          go mod tidy
      - name: Test with the Go CLI
        run: |
          cd backend
          go test ./...
      - name: Format code
        run: |
          cd backend
          ./scripts/format.sh
      - name: Build
        env:
          GO_ENV: production
          GOOS: linux
          GOARCH: amd64
        run: |
          cd backend
          go build -ldflags="-s -w"

  frontend:
    runs-on: ubuntu-latest

    strategy:
      matrix:
        node-version: [10.x, 12.x, 14.x, 15.x]

    steps:
      - uses: actions/checkout@v3
      - name: Use Node.js ${{ matrix.node-version }}
        uses: actions/setup-node@v3
        with:
          node-version: ${{ matrix.node-version }}
      - run: |
          cd frontend
          npm install
      - name: Format
      - run: |
          cd frontend
          npm run format
      - name: Test
      - run: |
          cd frontend
          npm test
